name: Build APK from Zip

on:
  push:
    branches: [ main, master ]
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up JDK 17
        uses: actions/setup-java@v4
        with:
          distribution: 'temurin'
          java-version: '17'

      - name: Find and extract project zip
        run: |
          ZIP_FILE=$(find . -maxdepth 1 -iname "*.zip" | head -n 1)
          if [ -z "$ZIP_FILE" ]; then
            echo "No zip file found in repo root. Skipping extraction."
            exit 0
          fi
          echo "Found zip: $ZIP_FILE"
          mkdir -p extracted
          unzip -o "$ZIP_FILE" -d extracted

      - name: Locate project folder
        run: |
          PROJECT_DIR=$(find extracted -maxdepth 4 -name "settings.gradle" -exec dirname {} \; | head -n 1)
          if [ -z "$PROJECT_DIR" ]; then
            PROJECT_DIR="."
          fi
          echo "PROJECT_DIR=$PROJECT_DIR" >> "$GITHUB_ENV"
          echo "Using project dir: $PROJECT_DIR"

      - name: Generate Gradle wrapper jar
        working-directory: ${{ env.PROJECT_DIR }}
        run: gradle wrapper --gradle-version 8.2

      - name: Build Debug APK
        working-directory: ${{ env.PROJECT_DIR }}
        run: |
          chmod +x gradlew
          ./gradlew assembleDebug

      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: Calculator-debug-apk
          path: ${{ env.PROJECT_DIR }}/app/build/outputs/apk/debug/app-debug.apk
